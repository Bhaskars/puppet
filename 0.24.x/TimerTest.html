def test_timer
  @timer.on_alarm do
    puts "[#{@timer.time_left} seconds left after alarm]"
    EventLoop.quit
  end
  8.times do
    t0 = Time.now
    @timer.start ; EventLoop.run
    t1 = Time.now
    assert(t1 - t0 > @timer.interval - @timer.tolerance)
  end
end

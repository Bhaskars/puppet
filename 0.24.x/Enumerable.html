def count(*args)
  seq = 0
  if !args.empty?
    item = args[0]
    each { |o| seq += 1 if item == o }
  elsif block_given?
    each { |o| seq += 1 if yield(o) }
  else
    each { seq += 1 }
  end
  seq
end

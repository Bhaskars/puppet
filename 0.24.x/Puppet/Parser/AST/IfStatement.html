def evaluate(scope)
    value = @test.safeevaluate(scope)

    if Puppet::Parser::Scope.true?(value)
        return @statements.safeevaluate(scope)
    else
        if defined? @else
            return @else.safeevaluate(scope)
        else
            return nil
        end
    end
end

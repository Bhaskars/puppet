def self.side(side = nil)
    if side
        side = side.intern if side.is_a?(String)
        unless [:client, :server].include?(side)
            raise ArgumentError, "Invalid side registration '%s' for %s" % [side, self.name]
        end
        @side = side
    else
        @side ||= :server
        return @side
    end
end

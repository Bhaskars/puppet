def self.store(name)
    name = symbolize(name)
    unless @stores.include? name
        if @loader.load(name)
            unless @stores.include? name
                Puppet.warning(
                    "Loaded report file for %s but report was not defined" %
                    name
                )
                return nil
            end
        else
            return nil
        end
    end
    @stores[name]
end

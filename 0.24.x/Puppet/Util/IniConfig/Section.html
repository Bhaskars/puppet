def []=(key, value)
    entry = find_entry(key)
    @dirty = true
    if entry.nil?
        @entries << [key, value]
    else
        entry[1] = value
    end
end

def format
    text = "[#{name}]\n"
    @entries.each do |entry|
        if entry.is_a?(Array)
            key, value = entry
            unless value.nil?
                text << "#{key}=#{value}\n"
            end
        else
            text << entry
        end
    end
    return text
end

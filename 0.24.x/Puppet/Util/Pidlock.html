def lock(opts = {})
	opts = {:anonymous => false}.merge(opts)

	if locked?
		mine?
	else
		if opts[:anonymous]
			File.open(@lockfile, 'w') { |fd| true }
		else
			File.open(@lockfile, "w") { |fd| fd.write(Process.pid) }
		end
		true
	end
end

def unlock(opts = {})
	opts = {:anonymous => false}.merge(opts)

	if mine? or (opts[:anonymous] and anonymous?)
		File.unlink(@lockfile)
		true
	else
		false
	end
end

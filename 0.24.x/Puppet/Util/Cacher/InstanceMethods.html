def expire
    # Only expire if we have an expirer.  This is
    # mostly so that we can comfortably handle cases
    # like Puppet::Type instances, which use their
    # catalog as their expirer, and they often don't
    # have a catalog.
    if e = expirer
        e.expire
    end
end

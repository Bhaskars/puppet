def new_markup(str, remove_para=false)
  first = @markup.nil?
  res = old_markup(str, remove_para)
  if first and not @markup.nil?
    @markup.add_special(/\b([a-z]\w+(::\w+)*)/,:CROSSREF)
    # we need to call it again, since we added a rule
    res = old_markup(str, remove_para)
  end
  res
end

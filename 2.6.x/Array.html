def combination(num)
  return [] if num < 0 || num > size
  return [[]] if num == 0
  return map{|e| [e] } if num == 1
  tmp = self.dup
  self[0, size - (num - 1)].inject([]) do |ret, e|
    tmp.shift
    ret += tmp.combination(num - 1).map{|a| a.unshift(e) }
  end
end

def to_zaml(z)
  z.first_time_only(self) {
    z.nested {
      if empty?
        z.emit('[]')
      else
        each { |v| z.nl('- '); v.to_zaml(z) }
      end
    }
  }
end

def to_pson(state = nil, *)
  case
  when infinite?
    if !state || state.allow_nan?
      to_s
    else
      raise GeneratorError, "#{self} not allowed in PSON"
    end
  when nan?
    if !state || state.allow_nan?
      to_s
    else
      raise GeneratorError, "#{self} not allowed in PSON"
    end
  else
    to_s
  end
end

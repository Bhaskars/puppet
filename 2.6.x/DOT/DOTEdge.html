def to_s (t = '')
  t + "#{@from} #{edge_link} #{to} [\n" +
    @options.to_a.collect{ |i|
      i[1] && i[0] != 'label' ?
        t + $tab + "#{i[0]} = #{i[1]}" :
        i[1] ? t + $tab + "#{i[0]} = \"#{i[1]}\"" : nil
    }.compact.join( "\n" ) + "\n#{t}]\n"
end

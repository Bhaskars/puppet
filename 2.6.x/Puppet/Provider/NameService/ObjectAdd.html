def modifycmd(param, value)
  cmd = [command(param.to_s =~ /password_.+_age/ ? :password : :modify)]
  cmd << flag(param) << value
  if @resource.allowdupe? && ((param == :uid) || (param == :gid and self.class.name == :groupadd))
    cmd << "-o"
  end
  cmd << @resource[:name]

  cmd
end

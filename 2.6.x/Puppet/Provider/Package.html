def self.prefetch(packages)
  instances.each do |prov|
    if pkg = packages[prov.name]
      pkg.provider = prov
    end
  end
end

def properties
  if @property_hash.empty?
    @property_hash = query || {:ensure => :absent}
    @property_hash[:ensure] = :absent if @property_hash.empty?
  end
  @property_hash.dup
end

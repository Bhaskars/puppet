def install_accumulator(attribute)
  self.accumulators[attribute] = Base.new(self, attribute)
  module_eval %{
    def self.accumulate_by_#{attribute.to_s}(*keys)
      accumulators[:#{attribute.to_s}].find(*keys)
    end
  }
end

def logopts
  if master?
    {
      :default => "$vardir/log",
      :mode    => 0750,
      :owner   => "service",
      :group   => "service",
      :desc    => "The Puppet log directory."
    }
  else
    ["$vardir/log", "The Puppet log directory."]
  end
end

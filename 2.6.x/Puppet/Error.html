def to_s
  str = nil
  if self.file and self.line
    str = "#{@message} at #{@file}:#{@line}"
  elsif self.line
    str = "#{@message} at line #{@line}"
  elsif self.file
    str = "#{@message} in #{self.file}"
  else
    str = @message.to_s
  end

  str
end
